```bash
#!/bin/bash

if [ "$(uname)" == 'Darwin' ]  && [ "$(uname -m)" == 'x86_64' ]; then
  echo 'x86_64 Mac'
elif [ "$(uname)" == 'Darwin' ]  && [ "$(uname -m)" == 'arm64' ]; then
  echo 'arm64 Mac'
fi
```
